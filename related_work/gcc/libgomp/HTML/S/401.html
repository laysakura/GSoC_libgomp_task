#define N 1500

int foo(void)
{
  int i, j;
  int x[N][N], y[N];

  for (i = 0; i < N; i++)
    {
      y[i] = i;

      for (j = 0; j < N; j++)
        {
          if (j > 500)
            {
              x[i][j] = i + j + 3;
              y[j] = i*j + 10;
            }
          else
            x[i][j] = x[i][j]*3;
        }
    }

  return x[2][5]*y[8];
}

int main(void)
{
  foo();

  return 0;
}
