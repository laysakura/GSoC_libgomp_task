/* This file contains timer routines for mingw32.  */

#include "libgomp.h"
#include <unistd.h>
#include <sys/timeb.h>

double
omp_get_wtime (void)
{
  struct _timeb timebuf;
  _ftime (&timebuf);
  return (timebuf.time + (long)(timebuf.millitm) / 1e3);
}

double
omp_get_wtick (void)
{
  return 1e-3;
}

ialias (omp_get_wtime)
ialias (omp_get_wtick)
