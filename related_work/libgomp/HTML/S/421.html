/* This file handles the BARRIER construct.  */

#include "libgomp.h"


void
GOMP_barrier (void)
{
  struct gomp_thread *thr = gomp_thread ();
  struct gomp_team *team = thr->ts.team;

  /* It is legal to have orphaned barriers.  */
  if (team == NULL)
    return;

  gomp_team_barrier_wait (&team->barrier);
}
